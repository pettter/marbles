\makeatletter
\def\renewcounter#1{%
  \@ifundefined{c@#1}
    {\@latex@error{counter #1 undefined}\@ehc}%
  \relax
  \let\@ifdefinable\@rc@ifdefinable
  \@ifnextchar[{\@newctr{#1}}{}
}
\newcounter {chapter}
\renewcounter {section}[chapter]
\renewcounter {subsection}[section]

\let\localinput\input  % Use this command to include LaTeX files

\renewcommand{\title}[1]{%
  \refstepcounter{chapter}
  \global\edef\@thetitle{#1}%
}
\renewcommand{\author}[1]{%
  \bgroup
    \def\and{ and }
    \global\edef\@theauthors{#1}%
  \egroup
}

\let\abstract\undefined

\renewcommand{\maketitle}{%
  \vspace*{\bigskipamount}
  \bgroup
    \begin{flushright}
      \sffamily\bfseries\LARGE
      C\textsmaller[2]{HAPTER}~\thechapter
      \bigskip\par
	  \Huge\@thetitle
	  \par
	  \smallskip
	  \Large
	  by \@theauthors
    \end{flushright}
  \egroup
  \vspace{1cm}  \addtocontents{toc}{\protect\@lectheading{\thechapter}{\@thetitle}{\@theauthors}}%
}

\newcommand{\@lectheading}[3]{%
  \vspace{2\bigskipamount}\par
  \hbox to \linewidth{\hspace*{\fill}Chapter #1\hspace*{\fill}}\nopagebreak\vspace{\smallskipamount}\nopagebreak\par\nopagebreak
  \hbox to \linewidth{\hspace*{\fill}\bfseries #2\hspace*{\fill}}\nopagebreak\par\nopagebreak
  \hbox to \linewidth{\hspace*{\fill}\bfseries by #3\hspace*{\fill}}\nopagebreak\par\nopagebreak
}%
